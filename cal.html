import React, { useState, useMemo, useEffect } from 'react';

import { 

  Search, ArrowUpDown, Plus, Trash2, Settings, LayoutDashboard, 

  FileSpreadsheet, CheckCircle2, AlertCircle, Percent, DollarSign,

  Link, DownloadCloud, Loader2, RefreshCw

} from 'lucide-react';


// สีหลักของ Theme (Pink F96096)

const themeColor = '#F96096';


export default function App() {

  const [activeTab, setActiveTab] = useState('dashboard');

  const [searchQuery, setSearchQuery] = useState('');

  const [sortAsc, setSortAsc] = useState(true);

  const [products, setProducts] = useState([]);

  const [isLoading, setIsLoading] = useState(false);

  const [importStatus, setImportStatus] = useState(null);


  // ลิงก์ Google Sheet จากผู้ใช้

  const sheetUrl = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vTO1G-j9WY2a_dzGOIDt_ei0ILzcfAM_Jq0p8MqohoUQY9ht0mLakj0KmjHoJQCP4CCLIFgQPNSuweR/pubhtml?gid=507359461&single=true';


  // ตั้งค่าช่องทางการขายเริ่มต้น

  const [channels, setChannels] = useState([

    {

      id: 1,

      name: 'Shopee',

      fees: [

        { id: 1, name: 'Service Fee', type: 'percent', value: 5.35 },

        { id: 2, name: 'Transaction Fee', type: 'percent', value: 3.21 }

      ]

    },

    {

      id: 2,

      name: 'TikTok Shop',

      fees: [

        { id: 1, name: 'Commission Fee', type: 'percent', value: 4.0 },

        { id: 2, name: 'Transaction Fee', type: 'percent', value: 3.0 }

      ]

    }

  ]);


  // ฟังก์ชันแยกข้อมูล CSV แบบรองรับ Quote

  const parseCSV = (text) => {

    const rows = [];

    let currentRow = [];

    let currentCell = '';

    let inQuotes = false;

    for (let i = 0; i < text.length; i++) {

      const char = text[i];

      if (char === '"' && text[i+1] === '"') {

        currentCell += '"'; i++;

      } else if (char === '"') {

        inQuotes = !inQuotes;

      } else if (char === ',' && !inQuotes) {

        currentRow.push(currentCell); currentCell = '';

      } else if (char === '\n' && !inQuotes) {

        currentRow.push(currentCell); rows.push(currentRow); currentRow = []; currentCell = '';

      } else if (char !== '\r') {

        currentCell += char;

      }

    }

    if (currentCell || currentRow.length) {

      currentRow.push(currentCell); rows.push(currentRow);

    }

    return rows;

  };


  // ฟังก์ชันดึงข้อมูลจาก Google Sheet

  const fetchSheetData = async () => {

    setIsLoading(true);

    setImportStatus(null);

    try {

      // แปลงลิงก์ pubhtml เป็น CSV

      const csvUrl = sheetUrl.replace('/pubhtml', '/pub') + (sheetUrl.includes('?') ? '&' : '?') + 'output=csv';

      

      const response = await fetch(csvUrl);

      if (!response.ok) throw new Error('Network response was not ok');

      const text = await response.text();

      

      const rows = parseCSV(text);

      if (rows.length < 2) throw new Error('No data found');


      // อ่านหัวตาราง (Headers)

      const headers = rows[0].map(h => h.trim().toLowerCase());

      

      // หาตำแหน่งคอลัมน์ตามที่ระบุ

      const idxBrand = headers.indexOf('brand');

      const idxName = headers.indexOf('name');

      const idxSku = headers.indexOf('sku');

      const idxSize = headers.indexOf('size');

      const idxCost = headers.indexOf('net_cost');

      const idxPrice = headers.indexOf('price_main');


      const mappedData = [];

      for (let i = 1; i < rows.length; i++) {

        const row = rows[i];

        if (row.length < 2) continue;


        const getVal = (idx) => idx >= 0 && row[idx] ? row[idx].trim() : '';

        const costVal = parseFloat(getVal(idxCost).replace(/,/g, '')) || 0;

        const priceVal = parseFloat(getVal(idxPrice).replace(/,/g, '')) || 0;


        mappedData.push({

          id: `item-${i}-${Date.now()}`,

          brand: getVal(idxBrand) || 'ไม่ระบุแบรนด์',

          name: getVal(idxName) || 'ไม่มีชื่อสินค้า',

          option: getVal(idxSku) || '-',

          size: getVal(idxSize) || '-',

          cost: costVal,

          price: priceVal

        });

      }


      setProducts(mappedData);

      setImportStatus({ type: 'success', message: `ดึงข้อมูลสำเร็จ ${mappedData.length} รายการ` });

    } catch (error) {

      console.error('Fetch Error:', error);

      setImportStatus({ type: 'error', message: 'ไม่สามารถดึงข้อมูลได้ โปรดตรวจสอบการแชร์ไฟล์ Google Sheet' });

    } finally {

      setIsLoading(false);

    }

  };


  // ดึงข้อมูลครั้งแรกเมื่อโหลดแอป

  useEffect(() => {

    fetchSheetData();

  }, []);


  // ระบบค้นหาและจัดเรียง

  const processedProducts = useMemo(() => {

    let result = [...products];

    if (searchQuery) {

      const q = searchQuery.toLowerCase();

      result = result.filter(p => 

        p.brand.toLowerCase().includes(q) || 

        p.name.toLowerCase().includes(q) ||

        p.option.toLowerCase().includes(q)

      );

    }


    result.sort((a, b) => {

      const brandComp = a.brand.localeCompare(b.brand, 'th');

      if (brandComp !== 0) return sortAsc ? brandComp : -brandComp;

      return sortAsc ? a.name.localeCompare(b.name, 'th') : -a.name.localeCompare(b.name, 'th');

    });


    return result;

  }, [products, searchQuery, sortAsc]);


  // คำนวณค่าธรรมเนียมและกำไร

  const calculateMetrics = (price, cost, channel) => {

    let totalPct = 0;

    let totalFixed = 0;

    channel.fees.forEach(f => {

      if (f.type === 'percent') totalPct += parseFloat(f.value || 0);

      else totalFixed += parseFloat(f.value || 0);

    });


    const feeAmount = (price * (totalPct / 100)) + totalFixed;

    const netIncome = price - feeAmount;

    const profit = netIncome - cost;

    const profitMargin = price > 0 ? (profit / price) * 100 : 0;


    return { netIncome, profit, profitMargin };

  };


  const formatNum = (n) => n.toLocaleString('th-TH', { minimumFractionDigits: 2, maximumFractionDigits: 2 });


  // จัดการช่องทางขาย

  const addChannel = () => setChannels([...channels, { id: Date.now(), name: 'ช่องทางใหม่', fees: [] }]);

  const removeChannel = (id) => setChannels(channels.filter(c => c.id !== id));

  const updateChannelName = (id, name) => setChannels(channels.map(c => c.id === id ? { ...c, name } : c));

  const addFee = (cid) => setChannels(channels.map(c => c.id === cid ? { ...c, fees: [...c.fees, { id: Date.now(), name: 'ค่าธรรมเนียม', type: 'percent', value: 0 }] } : c));

  const updateFee = (cid, fid, field, val) => setChannels(channels.map(c => c.id === cid ? { ...c, fees: c.fees.map(f => f.id === fid ? { ...f, [field]: val } : f) } : c));

  const removeFee = (cid, fid) => setChannels(channels.map(c => c.id === cid ? { ...c, fees: c.fees.filter(f => f.id !== fid) } : c));


  return (

    <div className="min-h-screen bg-pink-50 font-sans text-gray-800 pb-10">

      <nav className="bg-white sticky top-0 z-10 shadow-sm border-b border-pink-100">

        <div className="max-w-7xl mx-auto px-4 flex justify-between h-16 items-center">

          <div className="flex items-center space-x-2">

            <div className="w-10 h-10 rounded-xl flex items-center justify-center text-white shadow-lg" style={{ backgroundColor: themeColor }}>

              <DollarSign size={24} />

            </div>

            <h1 className="text-xl font-bold tracking-tight" style={{ color: themeColor }}>E-Com Pricing</h1>

          </div>

          <div className="flex bg-pink-50 p-1 rounded-lg">

            <button onClick={() => setActiveTab('dashboard')} className={`px-4 py-2 rounded-md text-sm font-medium transition-all ${activeTab === 'dashboard' ? 'bg-white text-[#F96096] shadow-sm' : 'text-gray-500 hover:text-[#F96096]'}`}>

              Dashboard

            </button>

            <button onClick={() => setActiveTab('channels')} className={`px-4 py-2 rounded-md text-sm font-medium transition-all ${activeTab === 'channels' ? 'bg-white text-[#F96096] shadow-sm' : 'text-gray-500 hover:text-[#F96096]'}`}>

              ตั้งค่าธรรมเนียม

            </button>

          </div>

        </div>

      </nav>


      <main className="max-w-7xl mx-auto px-4 mt-8">

        {activeTab === 'dashboard' && (

          <div className="space-y-6">

            {/* Toolbar */}

            <div className="bg-white p-4 rounded-2xl shadow-sm border border-pink-100 flex flex-col md:flex-row gap-4 items-center justify-between">

              <div className="relative w-full md:w-96">

                <Search className="absolute left-3 top-1/2 -translate-y-1/2 text-gray-400" size={18} />

                <input 

                  type="text" 

                  placeholder="ค้นหาแบรนด์ หรือ ชื่อสินค้า..." 

                  className="w-full pl-10 pr-4 py-2 bg-pink-50/50 border-none rounded-xl focus:ring-2 focus:ring-[#F96096] text-sm"

                  value={searchQuery}

                  onChange={(e) => setSearchQuery(e.target.value)}

                />

              </div>

              <div className="flex gap-2 w-full md:w-auto">

                <button onClick={fetchSheetData} className="flex-1 md:flex-none flex items-center justify-center px-4 py-2 bg-white border border-pink-200 text-[#F96096] rounded-xl hover:bg-pink-50 transition-colors text-sm font-medium">

                  {isLoading ? <Loader2 className="mr-2 animate-spin" size={18} /> : <RefreshCw className="mr-2" size={18} />}

                  รีเฟรชข้อมูล

                </button>

                <button onClick={() => setSortAsc(!sortAsc)} className="flex-1 md:flex-none flex items-center justify-center px-4 py-2 bg-[#F96096] text-white rounded-xl shadow-md shadow-pink-200 hover:opacity-90 transition-all text-sm font-medium">

                  <ArrowUpDown className="mr-2" size={18} />

                  เรียง: {sortAsc ? 'แบรนด์ ก-ฮ' : 'แบรนด์ ฮ-ก'}

                </button>

              </div>

            </div>


            {/* Status Info */}

            {importStatus && (

              <div className={`p-4 rounded-xl flex items-center text-sm font-medium border ${importStatus.type === 'success' ? 'bg-green-50 text-green-700 border-green-100' : 'bg-red-50 text-red-700 border-red-100'}`}>

                {importStatus.type === 'success' ? <CheckCircle2 size={18} className="mr-2" /> : <AlertCircle size={18} className="mr-2" />}

                {importStatus.message}

              </div>

            )}


            {/* Dashboard Table */}

            <div className="bg-white rounded-2xl shadow-sm border border-pink-100 overflow-hidden">

              <div className="overflow-x-auto">

                <table className="w-full border-collapse">

                  <thead>

                    <tr className="bg-pink-50/50 border-b border-pink-100">

                      <th colSpan="4" className="px-6 py-4 text-left text-xs font-bold text-gray-500 uppercase tracking-wider border-r border-pink-100">ข้อมูลสินค้า</th>

                      <th className="px-6 py-4 text-right text-xs font-bold text-gray-500 uppercase tracking-wider border-r border-pink-100">ต้นทุน/ขาย</th>

                      {channels.map(c => (

                        <th key={c.id} colSpan="3" className="px-6 py-4 text-center text-xs font-bold uppercase tracking-wider border-r last:border-r-0 border-pink-100" style={{ color: themeColor }}>

                          {c.name}

                        </th>

                      ))}

                    </tr>

                    <tr className="bg-white border-b border-pink-50 text-[11px] font-bold text-gray-400 uppercase">

                      <th className="px-6 py-2 text-left">แบรนด์</th>

                      <th className="px-6 py-2 text-left">ชื่อสินค้า</th>

                      <th className="px-6 py-2 text-left">ตัวเลือก</th>

                      <th className="px-6 py-2 text-left border-r border-pink-50">จำนวน</th>

                      <th className="px-6 py-2 text-right border-r border-pink-50">Cost | Price</th>

                      {channels.map(c => (

                        <React.Fragment key={`sub-${c.id}`}>

                          <th className="px-4 py-2 text-right bg-pink-50/20">เงินสุทธิ</th>

                          <th className="px-4 py-2 text-right bg-pink-50/20">กำไร(฿)</th>

                          <th className="px-4 py-2 text-right bg-pink-50/20 border-r last:border-r-0 border-pink-50">กำไร(%)</th>

                        </React.Fragment>

                      ))}

                    </tr>

                  </thead>

                  <tbody className="divide-y divide-pink-50">

                    {processedProducts.length === 0 ? (

                      <tr><td colSpan="100" className="px-6 py-20 text-center text-gray-400 italic">ไม่พบข้อมูล หรือกำลังโหลด...</td></tr>

                    ) : (

                      processedProducts.map(p => (

                        <tr key={p.id} className="hover:bg-pink-50/30 transition-colors">

                          <td className="px-6 py-4 whitespace-nowrap text-sm font-semibold text-gray-700">{p.brand}</td>

                          <td className="px-6 py-4 whitespace-nowrap text-sm text-gray-600 max-w-[200px] truncate" title={p.name}>{p.name}</td>

                          <td className="px-6 py-4 whitespace-nowrap text-xs text-gray-500">{p.option}</td>

                          <td className="px-6 py-4 whitespace-nowrap text-xs text-gray-500 border-r border-pink-50">{p.size}</td>

                          <td className="px-6 py-4 whitespace-nowrap text-right border-r border-pink-50">

                            <div className="text-[10px] text-gray-400">ต้นทุน: {formatNum(p.cost)}</div>

                            <div className="text-sm font-bold text-gray-800 underline decoration-pink-300 decoration-2 underline-offset-4">{formatNum(p.price)}</div>

                          </td>

                          {channels.map(c => {

                            const { netIncome, profit, profitMargin } = calculateMetrics(p.price, p.cost, c);

                            const isLoss = profit < 0;

                            return (

                              <React.Fragment key={`cell-${p.id}-${c.id}`}>

                                <td className="px-4 py-4 text-right text-sm font-bold text-blue-600 bg-blue-50/10">{formatNum(netIncome)}</td>

                                <td className={`px-4 py-4 text-right text-sm font-bold ${isLoss ? 'text-red-500' : 'text-green-600'} bg-pink-50/10`}>

                                  {formatNum(profit)}

                                </td>

                                <td className={`px-4 py-4 text-right text-xs font-black border-r last:border-r-0 border-pink-50 ${isLoss ? 'text-red-600 bg-red-50/30' : 'text-green-700 bg-green-50/30'}`}>

                                  {profitMargin.toFixed(1)}%

                                </td>

                              </React.Fragment>

                            );

                          })}

                        </tr>

                      ))

                    )}

                  </tbody>

                </table>

              </div>

            </div>

          </div>

        )}


        {activeTab === 'channels' && (

          <div className="max-w-3xl mx-auto space-y-6">

            <div className="flex justify-between items-center bg-white p-6 rounded-2xl shadow-sm border border-pink-100">

              <h2 className="text-xl font-bold text-gray-800 flex items-center">

                <Settings className="mr-2" style={{ color: themeColor }} /> ตั้งค่าช่องทางและค่าธรรมเนียม

              </h2>

              <button onClick={addChannel} className="px-4 py-2 text-sm font-bold text-white rounded-xl shadow-lg hover:opacity-90 transition-all flex items-center" style={{ backgroundColor: themeColor }}>

                <Plus size={18} className="mr-1" /> เพิ่มช่องทาง

              </button>

            </div>


            {channels.map((c, idx) => (

              <div key={c.id} className="bg-white rounded-2xl shadow-sm border border-pink-100 overflow-hidden">

                <div className="bg-pink-50/50 px-6 py-4 flex justify-between items-center border-b border-pink-100">

                  <div className="flex items-center gap-3">

                    <span className="w-8 h-8 rounded-lg bg-white border border-pink-200 flex items-center justify-center font-bold text-xs" style={{ color: themeColor }}>{idx + 1}</span>

                    <input 

                      type="text" 

                      value={c.name} 

                      onChange={(e) => updateChannelName(c.id, e.target.value)}

                      className="bg-transparent font-bold border-none focus:ring-0 p-0 text-gray-700 text-lg"

                    />

                  </div>

                  <button onClick={() => removeChannel(c.id)} className="text-red-400 hover:text-red-600 transition-colors">

                    <Trash2 size={18} />

                  </button>

                </div>

                <div className="p-6 space-y-4">

                  {c.fees.map(f => (

                    <div key={f.id} className="flex flex-col sm:flex-row gap-4 items-center bg-gray-50/50 p-3 rounded-xl border border-gray-100">

                      <input 

                        type="text" 

                        value={f.name} 

                        onChange={(e) => updateFee(c.id, f.id, 'name', e.target.value)}

                        placeholder="ชื่อค่าธรรมเนียม" 

                        className="flex-1 text-sm bg-white border-gray-200 rounded-lg focus:ring-[#F96096] focus:border-[#F96096]"

                      />

                      <div className="flex bg-white rounded-lg p-1 border border-gray-200">

                        <button onClick={() => updateFee(c.id, f.id, 'type', 'percent')} className={`px-3 py-1 text-xs font-bold rounded-md transition-all ${f.type === 'percent' ? 'bg-[#F96096] text-white shadow-md' : 'text-gray-400'}`}>%</button>

                        <button onClick={() => updateFee(c.id, f.id, 'type', 'fixed')} className={`px-3 py-1 text-xs font-bold rounded-md transition-all ${f.type === 'fixed' ? 'bg-[#F96096] text-white shadow-md' : 'text-gray-400'}`}>฿</button>

                      </div>

                      <div className="w-full sm:w-32">

                        <input 

                          type="number" 

                          value={f.value} 

                          onChange={(e) => updateFee(c.id, f.id, 'value', e.target.value)}

                          className="w-full text-right text-sm border-gray-200 rounded-lg focus:ring-[#F96096]"

                        />

                      </div>

                      <button onClick={() => removeFee(c.id, f.id)} className="text-gray-300 hover:text-red-500">

                        <Trash2 size={16} />

                      </button>

                    </div>

                  ))}
